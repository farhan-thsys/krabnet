
3. Update `KrabnetServer::new()` to set `tier3_sender: None`.

4. Update `KrabnetServer::with_wal()` to set `tier3_sender: None`.

5. Add a new constructor method `with_wal_and_tier3()` that accepts both a WAL writer and a Tier3Sender:
   ```rust
   pub fn with_wal_and_tier3(
       engine: Arc<RwLock<Engine>>,
       wal_writer: Arc<Mutex<WalWriter>>,
       tier3_sender: Tier3Sender,
   ) -> Self {
       let (frame_tx, _) = broadcast::channel(1024);
       Self {
           engine,
           frame_tx,
           wal_writer: Some(wal_writer),
           tier3_sender: Some(tier3_sender),
       }
   }
   ```

6. In the `ingest_event()` method, AFTER the engine.ingest() call and WAL persistence, add broadcast and Tier 3 dispatch:

   a. **Broadcast**: Query the engine (read lock) for all frames via `list_frames()`. For each frame, query its current paths via `query_frame()`. Build a `proto::FrameUpdate { frame_id, paths: paths_to_proto(&paths), epoch: epoch.0 }` and send via `self.frame_tx.send(update)`. Ignore send errors (no subscribers is OK — `send()` returns `Err` when receiver count is 0, which is fine).

   NOTE: Instead of broadcasting ALL frames, be efficient — only broadcast frames that were potentially affected. Since we don't have direct access to affected_frames from the gRPC layer (the engine's ingest handles routing internally), broadcast ALL registered frames. The subscribe_frame stream already filters by frame_id, so subscribers only receive updates for their subscribed frame. Use `engine.list_frames()` to get frame IDs, then `engine.query_frame(fid)` for each to get current paths. The engine must be re-locked for this (acquire write lock since query_frame takes &mut self).

   b. **Tier 3 dispatch**: For each frame that was queried, construct a `Tier2Result`:
   ```rust
   Tier2Result {
       frame_id: fid,
       anchor: anchor,
       paths: paths.clone(),
       epoch: epoch,
       tier2_summary: format!("{} paths materialized", paths.len()),
   }
   ```
   Send via `self.tier3_sender.as_ref().map(|s| s.try_send(result))`. The try_send is non-blocking per TIER3-04.

   The complete post-ingest block should look like:
   ```rust
   // Post-ingest: broadcast FrameUpdates and dispatch Tier 3 results
   {
       let mut engine = self.engine.write()
           .map_err(|_| Status::internal("engine lock poisoned"))?;
       let frames = engine.list_frames();
       for (fid, anchor, _tier, _tuple_count) in &frames {
           if let Some(paths) = engine.query_frame(*fid) {
               // Broadcast FrameUpdate for SubscribeFrame clients (GRPC-03)
               let update = proto::FrameUpdate {
                   frame_id: *fid,
                   paths: paths_to_proto(&paths),
                   epoch: epoch.0,
               };
               let _ = self.frame_tx.send(update);

               // Dispatch Tier 2 result to Tier 3 worker (TIER3-01..04)
               if let Some(ref sender) = self.tier3_sender {
                   let result = Tier2Result {
                       frame_id: *fid,
                       anchor: *anchor,
                       paths,
                       epoch,
                       tier2_summary: format!("{} paths materialized", paths.len()),
                   };
                   let _ = sender.try_send(result);
               }
           }
       }
   }
   ```

Here is the detailed test structure:

```rust
#[tokio::test]
async fn test_ingest_broadcasts_and_tier3() {
    use crate::tier3::{MockLlmClient, Tier3Worker};
    use std::time::Duration;

    // 1. Create engine
    let engine = Arc::new(RwLock::new(Engine::new(64)));

    // 2. Create Tier 3 worker with mock LLM
    let mock = MockLlmClient::new(vec!["test-analysis".to_string()]);
    let (worker, tier3_sender) = Tier3Worker::new(Box::new(mock));
    let tier3_results = worker.results_handle();

    // 3. Spawn worker thread
    let worker_handle = std::thread::spawn(move || {
        worker.run();
    });

    // 4. Create server with Tier3Sender
    let server = KrabnetServer::with_tier3(Arc::clone(&engine), tier3_sender);

    // 5. Start gRPC server
    let listener = tokio::net::TcpListener::bind("[::1]:0").await.unwrap();
    let addr = listener.local_addr().unwrap();
    let svc = server.into_service();
    let server_handle = tokio::spawn(async move {
        tonic::transport::Server::builder()
            .add_service(svc)
            .serve_with_incoming(tokio_stream::wrappers::TcpListenerStream::new(listener))
            .await
            .unwrap();
    });

    tokio::time::sleep(Duration::from_millis(50)).await;

    // 6. Connect client
    let mut client = KrabnetServiceClient::connect(
        format!("http://[::1]:{}", addr.port())
    ).await.unwrap();

    // 7. Ingest nodes + edge
    client.ingest_event(IngestEventRequest {
        event: Some(ingest_event_request::Event::NodeAdded(NodeAddedEvent {
            node_id: 1, type_id: 10,
        })),
    }).await.unwrap();

    client.ingest_event(IngestEventRequest {
        event: Some(ingest_event_request::Event::NodeAdded(NodeAddedEvent {
            node_id: 2, type_id: 20,
        })),
    }).await.unwrap();

    let edge_resp = client.ingest_event(IngestEventRequest {
        event: Some(ingest_event_request::Event::EdgeAdded(EdgeAddedEvent {
            edge_id: 0, source: 1, target: 2, type_id: 100,
        })),
    }).await.unwrap();
    let reg_epoch = edge_resp.into_inner().epoch;

    // 8. Register frame
    let reg_resp = client.register_frame(RegisterFrameRequest {
        anchor_node_id: 1,
        pattern: vec![proto::HopSpec {
            direction: Direction::Outgoing as i32,
            edge_type: Some(100),
            target_type: Some(20),
            filter: Some(proto::Filter {
                filter_type: Some(filter::FilterType::None(true)),
            }),
        }],
        epoch: reg_epoch,
    }).await.unwrap();
    let frame_id = reg_resp.into_inner().frame_id;

    // 9. Subscribe to frame updates BEFORE triggering event
    let mut subscribe_stream = client
        .subscribe_frame(SubscribeFrameRequest { frame_id })
        .await
        .unwrap()
        .into_inner();

    // 10. Ingest another event to trigger broadcast
    client.ingest_event(IngestEventRequest {
        event: Some(ingest_event_request::Event::NodeAdded(NodeAddedEvent {
            node_id: 3, type_id: 30,
        })),
    }).await.unwrap();

    // 11. Wait for FrameUpdate on subscribe stream
    let update = tokio::time::timeout(
        Duration::from_secs(2),
        subscribe_stream.message(),
    ).await
        .expect("timeout waiting for FrameUpdate")
        .expect("stream error")
        .expect("stream ended");

    assert_eq!(update.frame_id, frame_id);
    assert!(!update.paths.is_empty(), "FrameUpdate should have paths");

    // 12. Shutdown server (drops Tier3Sender via the server)
    server_handle.abort();

    // Give Tier 3 worker time to process and exit
    tokio::time::sleep(Duration::from_millis(100)).await;

    // 13. Verify Tier 3 received results
    // Worker thread exits when sender is dropped (server aborted)
    let _ = worker_handle.join();
    let results = tier3_results.lock().unwrap();
    assert!(!results.is_empty(), "Tier 3 should have processed at least one result");
    assert_eq!(results[0].frame_id, frame_id);
}
```
